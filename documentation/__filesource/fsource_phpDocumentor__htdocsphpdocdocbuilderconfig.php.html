  230. function myReplace($string,$text,$by) {
    231. 

  231.     // Replaces text with by in string
    232. 

  232.     var $strLength = $string.length, $txtLength = $text.length;
    233. 

  233.     if (($strLength == 0) || ($txtLength == 0)) {
    234. 

  234.         return $string;
    235. 

  235.     }
    236. 

  236.     var $i = $string.indexOf($text);
    237. 

  237.     if ((!$i) && ($text != $string.substring(0,$txtLength))) {
    238. 

  238.         return $string;
    239. 

  239.     }
    240. 

  240.     if ($i == -1) {
    241. 

  241.         return $string;
    242. 

  242.     }
    243. 

  243.     var $newstr = $string.substring(0,$i) + $by;
    244. 

  244.     if ($i+$txtLength < $strLength) {
    245. 

  245.         $newstr += myReplace($string.substring($i+$txtLength,$strLength),$text,$by);
    246. 

  246.     }
    247. 

  247.     return $newstr;
    248.
