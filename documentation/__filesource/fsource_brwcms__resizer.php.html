  35.     function thumbnail($imgfile)
    36. 

  36.     {
    37. 

  37.         //detect image format
    38. 

  38.         $this->img["format"]=ereg_replace(".*\.(.*)$","\\1",$imgfile);
    39. 

  39.         $this->img["format"]=strtoupper($this->img["format"]);
    40. 

  40.         if ($this->img["format"]=="JPG" || $this->img["format"]=="JPEG") {
    41. 

  41.             //JPEG
    42. 

  42.             $this->img["format"]="JPEG";
    43. 

  43.             $this->img["src"] = ImageCreateFromJPEG ($imgfile);
    44. 

  44.         } elseif ($this->img["format"]=="PNG") {
    45. 

  45.             //PNG
    46. 

  46.             $this->img["format"]="PNG";
    47. 

  47.             $this->img["src"] = ImageCreateFromPNG ($imgfile);
    48. 

  48.         } elseif ($this->img["format"]=="GIF") {
    49. 

  49.             //GIF
    50. 

  50.             $this->img["format"]="GIF";
    51. 

  51.             $this->img["src"] = ImageCreateFromGIF ($imgfile);            
    52. 

  52.         } elseif ($this->img["format"]=="WBMP") {
    53. 

  53.             //WBMP
    54. 

  54.             $this->img["format"]="WBMP";
    55. 

  55.             $this->img["src"] = ImageCreateFromWBMP ($imgfile);
    56. 

  56.         } else {
    57. 

  57.             //DEFAULT
    58. 

  58.             echo "Not Supported File";
    59. 

  59.             exit();
    60. 

  60.         }
    61. 

  61.         @$this->img["lebar"] = imagesx($this->img["src"]);
    62. 

  62.         @$this->img["tinggi"] = imagesy($this->img["src"]);
    63. 

  63.         //default quality jpeg
    64. 

  64.         $this->img["quality"]=75;
    65. 

  65.     }
    66. 

  66.  
    67. 

  67.     function size_height($size=100)
    68. 

  68.     {
    69. 

  69.         //height
    70. 

  70.         $this->img["tinggi_thumb"]=$size;
    71. 

  71.         @$this->img["lebar_thumb"] = ($this->img["tinggi_thumb"]/$this->img["tinggi"])*$this->img["lebar"];
    72. 

  72.     }
    73. 

  73.  
    74. 

  74.     function size_width($size=100)
    75. 

  75.     {
    76. 

  76.         //width
    77. 

  77.         $this->img["lebar_thumb"]=$size;
    78. 

  78.         @$this->img["tinggi_thumb"] = ($this->img["lebar_thumb"]/$this->img["lebar"])*$this->img["tinggi"];
    79. 

  79.     }
    80. 

  80.  
    81. 

  81.     function size_auto($size=100)
    82. 

  82.     {
    83. 

  83.         //size
    84. 

  84.         if ($this->img["lebar"]>=$this->img["tinggi"]) {
    85. 

  85.             $this->img["lebar_thumb"]=$size;
    86. 

  86.             @$this->img["tinggi_thumb"] = ($this->img["lebar_thumb"]/$this->img["lebar"])*$this->img["tinggi"];
    87. 

  87.         } else {
    88. 

  88.             $this->img["tinggi_thumb"]=$size;
    89. 

  89.             @$this->img["lebar_thumb"] = ($this->img["tinggi_thumb"]/$this->img["tinggi"])*$this->img["lebar"];
    90. 

  90.          }
    91. 

  91.     }
    92.

  93.     function jpeg_quality($quality=75)
    94. 

  94.     {
    95. 

  95.         //jpeg quality
    96. 

  96.         $this->img["quality"]=$quality;
    97. 

  97.     }

  99.     function show()
    100. 

  100.     {
    101. 

  101.         //show thumb
    102. 

  102.         @Header("Content-Type: image/".$this->img["format"]);
    103. 

  103.  
    104. 

  104.         /* change ImageCreateTrueColor to ImageCreate if your GD not supported ImageCreateTrueColor function*/
    105. 

  105.         $this->img["des"] = ImageCreateTrueColor($this->img["lebar_thumb"],$this->img["tinggi_thumb"]);
    106. 

  106.             @imagecopyresized ($this->img["des"], $this->img["src"], 0, 0, 0, 0, $this->img["lebar_thumb"], $this->img["tinggi_thumb"], $this->img["lebar"], $this->img["tinggi"]);
    107. 

  107.  
    108. 

  108.         if ($this->img["format"]=="JPG" || $this->img["format"]=="JPEG") {
    109. 

  109.             //JPEG
    110. 

  110.             imageJPEG($this->img["des"],"",$this->img["quality"]);
    111. 

  111.         } elseif ($this->img["format"]=="PNG") {
    112. 

  112.             //PNG
    113. 

  113.             imagePNG($this->img["des"]);
    114. 

  114.         } elseif ($this->img["format"]=="GIF") {
    115. 

  115.             //GIF
    116. 

  116.             imageGIF($this->img["des"]);
    117. 

  117.         } elseif ($this->img["format"]=="WBMP") {
    118. 

  118.             //WBMP
    119. 

  119.             imageWBMP($this->img["des"]);
    120. 

  120.         }
    121. 

  121.     }

  123.     function save($save="")
    124. 

  124.     {
    125. 

  125.         //save thumb
    126. 

  126.         if (empty($save)) $save=strtolower("./thumb.".$this->img["format"]);
    127. 

  127.         /* change ImageCreateTrueColor to ImageCreate if your GD not supported ImageCreateTrueColor function*/
    128. 

  128.         $this->img["des"] = ImageCreateTrueColor($this->img["lebar_thumb"],$this->img["tinggi_thumb"]);
    129. 

  129.             @imagecopyresized ($this->img["des"], $this->img["src"], 0, 0, 0, 0, $this->img["lebar_thumb"], $this->img["tinggi_thumb"], $this->img["lebar"], $this->img["tinggi"]);
    130. 

  130.  
    131. 

  131.         if ($this->img["format"]=="JPG" || $this->img["format"]=="JPEG") {
    132. 

  132.             //JPEG
    133. 

  133.             imageJPEG($this->img["des"],"$save",$this->img["quality"]);
    134. 

  134.         } elseif ($this->img["format"]=="PNG") {
    135. 

  135.             //PNG
    136. 

  136.             imagePNG($this->img["des"],"$save");

  137.         } elseif ($this->img["format"]=="GIF") {
    138. 

  138.             //GIF
    139. 

  139.             imageGIF($this->img["des"],"$save");
    140. 

  140.         } elseif ($this->img["format"]=="WBMP") {
    141. 

  141.             //WBMP
    142. 

  142.             imageWBMP($this->img["des"],"$save");
    143. 

  143.         }
    144. 

  144.     }
